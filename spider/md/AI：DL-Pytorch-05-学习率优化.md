# 1. 学习率优化方法

## 1.1 为什么需要进行学习率优化?直接使用固定值不好吗？

### 1.1.1 解释说明

目标是使得**loss**尽量小  

学习率过大问题：  

- 参数更新步长过大，可能会跳过最优解，导致模型无法收敛。  

- 可能导致模型损失函数值在最优解附近来回震荡，甚至发散。  

学习率过小问题：  

- 训练速度慢，步长短，迭代次数多  

- 容器陷入局部最优解，无法跳出   

- 如果太小，永远无法收敛   

在训练初期一般需要较大的学习率，加速收敛，在训练中后期，需要较小的学习率，来避免震荡。  

不同的参数也需要不同的学习率  

### 1.1.2 **代码演示**  
```python
import matplotlib
matplotlib.use('TkAgg')  # 不弹出窗口
import matplotlib.pyplot as plt
import torch

def loss_func(x_t):
    # y = 4*x_t^2
    return torch.pow(2 * x_t, 2)


def fun():
    weight = torch.tensor([2.], requires_grad=True)
    iter_rec, loss_rec, weight_rec = list(), list(), list()

    # 学习率：0.01,0.02，0.03,0.1，0.2，0.3，0.4
    # lr = 0.1	# 正常的梯度下降

    # lr = 0.125
    # 当学习率设置0.125 一下子求出一个最优解
    # weight=0 loss=0 在weight=0处梯度等于0 weight的值weight=weight-lr*weight.grad就不用更新了
    # 后续再多少次选代 都固定在最优点

    lr = 0.2
    # weight从2.0一下子跨过0点,到了左侧负数区域

    # lr = 0.3
    # 梯度越来越大 梯度爆炸

    max_iteration = 4
    for i in range(max_iteration):
        loss = loss_func(weight)  # 得出Loss值
        loss.backward()  # 计算weight的梯度

        print("Iter:{}, Weight:{:8}, Weight.grad:{:8}, loss:{:10}".format(
            i, weight.detach().numpy()[0], weight.grad.detach().numpy()[0], loss.item()))

        # 权重 列表
        weight_rec.append(weight.item())

        # 更新参数
        weight.data.sub_(lr * weight.grad)  # weight= weight - lr * weight.grad
        weight.grad.zero_()

        # 迭代次数 列表
        iter_rec.append(i)

        # 损失值列表，item为获取标量值
        loss_rec.append(loss.item())

    # 迭代次数-损失值 关系图
    plt.subplot(121).plot(iter_rec, loss_rec, '-ro')
    plt.grid()
    plt.xlabel("Iteration X")
    plt.ylabel("Loss value Y")

    # 绘制损失函数曲线
    weight_2 = torch.linspace(-3, 3, 100)
    loss_2 = loss_func(weight_2)
    plt.subplot(122).plot(weight_2.detach().numpy(), loss_2.detach().numpy(), label='y = 4*x^2')

    # 损失函数值
    loss_2_rec = [loss_func(torch.tensor(i)).item() for i in weight_rec]
    print("weight_rec --->", weight_rec)
    print("loss_2_rec --->", loss_2_rec)
    plt.subplot(122).plot(weight_rec, loss_2_rec, '-ro')
    plt.grid()
    plt.xlabel("Weight X")
    plt.ylabel("Loss value Y")

    # 绘制
    plt.legend()
    plt.show()

if __name__ == '__main__':
    fun()
```

## 1.2 等间隔学习率衰减  lr_scheduler.StepLR(optimizer,step_size,gamma=0.1)
### 1.2.1 基本原理和公式

每经过step_size个epoch后，重新计算一下学习率。更新公式如下：   

$lr = lr * gamma$  

### 1.2.2 代码说明

**代码**  
```python
import matplotlib

matplotlib.use('TkAgg')
from matplotlib import pyplot as plt

import torch


def step_lr():
    # 0. 参数初始化
    LR = 0.1  # 设置学习率初始化值为0.1
    iteration = 10
    max_epoch = 200
    # 1 初始化参数
    y_true = torch.tensor([0])
    x = torch.tensor([1.0])
    w = torch.tensor([1.0], requires_grad=True)
    # 2. 优化器
    optimizer = torch.optim.SGD([w], lr=LR, momentum=0.9)
    # 3. 设置学习率下降策略
    scheduler_lr = torch.optim.lr_scheduler.StepLR(optimizer, step_size=50, gamma=0.5)
    # 4. 获取学习率的值和当前的epoch
    lr_list, epoch_list = list(), list()

    for epoch in range(max_epoch):
        lr_list.append(scheduler_lr.get_last_lr())  # 获取当前lr
        epoch_list.append(epoch)  # 获取当前的epoch
        for i in range(iteration):  # 遍历每一个batch数据
            loss = ((w * x - y_true) ** 2) / 2.0  # 目标函数
            # 梯度清零
            optimizer.zero_grad()
            # 反向传播,计算梯度
            loss.backward()
            # 更新权重
            optimizer.step()
        # 更新下一个epoch的学习率
        scheduler_lr.step()

    # 5. 绘制学习率变化的曲线
    print("lr_list:", lr_list)
    print("epoch_list", epoch_list)
    plt.plot(epoch_list, lr_list, label="Step LR Scheduler")
    plt.xlabel("Epoch")
    plt.ylabel("Learning rate")
    plt.legend()
    plt.show()
    pass


if __name__ == '__main__':
    step_lr()
    pass

```

## 1.3 指定间隔学习率衰减 torch.optim.lr_scheduler.MultiStepLR(optimizer,milestones,gamma)
### 1.3.1 原理

初始学习率（initial learning rate）：训练开始时的学习率。  

衰减间隔（decay steps）：每隔多少个 epoch 或 step 进行一次衰减。  

衰减系数（decay rate）：每次衰减时乘以的系数，通常为 0.1 或 0.5。  

学习率调度器（learning rate scheduler）：用于在训练过程中动态调整学习率。  

### 1.3.2 代码说明

```python
def multi_step_lr():
    # 200轮epoch循环
    TOTAL_EPOCH = 200
    # 每一轮10次迭代
    TOTAL_ITER = 10
    x = torch.tensor([1.0], dtype=torch.float32)
    y_true = torch.tensor([0], dtype=torch.float32)
    weight = torch.tensor([1.0], requires_grad=True)
    optimizer = torch.optim.SGD(params=[weight], lr=0.1, momentum=0.9)
    milestones = [20, 30, 50, 100]
    lr_scheduler = torch.optim.lr_scheduler.MultiStepLR(optimizer=optimizer, milestones=milestones, gamma=0.5)
    # 汇总数据记录
    epoch_rec, lr_rec = list(), list()
    for cur_epoch in range(TOTAL_EPOCH):
        epoch_rec.append(cur_epoch)
        lr_rec.append(lr_scheduler.get_last_lr())
        for cur_iter in range(TOTAL_ITER):
            # 计算损失
            loss = ((y_true - weight * x) ** 2) / 2
            # 计算梯度
            optimizer.zero_grad()
            loss.backward()
            # 更新参数
            optimizer.step()
        # 更新学习率
        lr_scheduler.step()
    # 打印
    # 5. 绘制学习率变化的曲线
    print("lr_list:", lr_rec)
    print("epoch_list", epoch_rec)
    plt.plot(epoch_rec, lr_rec, label="Step LR Scheduler")
    plt.xlabel("Epoch")
    plt.ylabel("Learning rate")
    plt.legend()
    plt.show()
    pass
```

## 1.4 指数学习率衰减 torch.optim.lr_scheduler.ExponentialLR(optimizer,gamma)
### 1.4.1 基本原理

初始学习率（initial learning rate）：训练开始时的学习率。  

衰减系数（gamma）：每次更新时乘以的系数，通常为 0.95 或 0.9。  

学习率调度器（learning rate scheduler）：用于在训练过程中动态调整学习率。  

指数衰减公式：  

$lr = lr * gamma^{epoch}$  


### 1.3.2 代码说明
```python
def exp_lr():
    # 200轮epoch循环
    TOTAL_EPOCH = 200
    # 每一轮10次迭代
    TOTAL_ITER = 10
    x = torch.tensor([1.0], dtype=torch.float32)
    y_true = torch.tensor([0], dtype=torch.float32)
    weight = torch.tensor([1.0], requires_grad=True)
    optimizer = torch.optim.SGD(params=[weight], lr=0.1, momentum=0.9)
    lr_scheduler = torch.optim.lr_scheduler.ExponentialLR(optimizer=optimizer, gamma=0.95)
    # 汇总数据记录
    epoch_rec, lr_rec = list(), list()
    for cur_epoch in range(TOTAL_EPOCH):
        epoch_rec.append(cur_epoch)
        lr_rec.append(lr_scheduler.get_last_lr())
        for cur_iter in range(TOTAL_ITER):
            # 计算损失
            loss = ((y_true - weight * x) ** 2) / 2
            # 计算梯度
            optimizer.zero_grad()
            loss.backward()
            # 更新参数
            optimizer.step()
        # 更新学习率
        lr_scheduler.step()
    # 打印
    # 5. 绘制学习率变化的曲线
    print("lr_list:", lr_rec)
    print("epoch_list", epoch_rec)
    plt.plot(epoch_rec, lr_rec, label="Step LR Scheduler")
    plt.xlabel("Epoch")
    plt.ylabel("Learning rate")
    plt.legend()
    plt.show()
    pass
```

## 1.5 总结
| 方法 | 等间隔学习率衰减（Step Decay） | 指定间隔学习率衰减（Exponential Decay） | 指数学习率衰减（Exponential Moving Average Decay） |
|------|-------------------------------|----------------------------------------|--------------------------------------------------|
| **衰减方式** | 固定步长衰减 | 指定步长衰减 | 平滑指数衰减，历史平均考虑 |
| **实现难度** | 简单易实现 | 相对简单，容易调整 | 需要额外历史计算，较复杂 |
| **适用场景** | 大型数据集、较为简单的任务 | 对训练平稳性要求较高的任务 | 高精度训练，避免过快收敛 |
| **优点** | 直观，易于调试，适用于大批量数据 | 易于调试，稳定训练过程 | 平滑且考虑历史更新，收敛稳定性较强 |
| **缺点** | 学习率变化较大，可能跳过最优点 | 在某些情况下可能衰减过快，导致优化提前停滞 | 超参数调节较为复杂，可能需要更多的计算资源 |
